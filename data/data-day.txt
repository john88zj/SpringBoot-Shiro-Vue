                function mouseMove()
                {
                    try
                    {
                       var ua = navigator.userAgent.toLowerCase();
                       var event = window.event || arguments.callee.caller.arguments[0];

                        
                        if (ua.match(/Firefox/i)=="firefox")
                         {                           
                              if
                              (
                               event.layerX > 676 | 
                               event.layerX <= 0 |
                               event.layerY <= 0
                               )
                               {
                                   return;
                               }

                         }
                         else
                         {
                              if
                              (
                               event.offsetX > 676 | 
                               event.offsetX <= 0 |
                               event.offsetY <= 0
                               )
                               {
                                   return;
                               }
                           }
                                       
                        var divMoveHor = ge('divMoveHor');

                        var divDetailVer = ge('divDetailVer');

                        if(divMoveHor == null)

                            createDiv();
                            divMoveHor = ge('divMoveHor');
//                          divMoveVer = ge('divMoveVer');
                            divDetailVer = ge('divDetailVer');
                        }
                        
                        divMoveHor.style.left = event.clientX - 53;
                        divMoveHor.style.top = 0;
//                      divMoveVer.style.left = 0;
//                      divMoveVer.style.top = event.clientY - 29;
                           
                        if (ua.match(/Firefox/i)=="firefox")
                        {
                        currentPositionX = event.layerY;//offsetY鼠标在当前div中的坐标，clientY鼠标在整个屏幕中的坐标
                        currnetPositionY = event.layerX;
                        }
                        else
                        {
                        currentPositionX = event.offsetY;//offsetY鼠标在当前div中的坐标，clientY鼠标在整个屏幕中的坐标
                        currnetPositionY = event.offsetX;
                        }     
                        
                        if(currentPositionX < 250)//鼠标与上边框的距离不足div的高度
                        
                        {
                            divDetailVer.style.top = currentPositionX + 50;//div出现在鼠标下方

                        else
                        {
                            divDetailVer.style.top = currentPositionX - 250;;//否则，div出现在鼠标上方
                            
                        }
                        if(currnetPositionY < 150)//鼠标与左边框的距离不足div的宽度
                        
                        {
                            divDetailVer.style.left = currnetPositionY + 50;//div出现在鼠标右方
                        }
                        else
                        {
                            divDetailVer.style.left = currnetPositionY-150;
                        }
                        
                        var aa;                        
                        if (ua.match(/Firefox/i)=="firefox")
                        {
                            aa = dataObjs[event.layerX];
                        }
                        else
                        {
                            aa = dataObjs[event.offsetX];
                        }  
                                         
                        var str_table = "";
                        if(Is_Apple() == true||ua.match(/Firefox/i)=="firefox"){
                            str_table = "<table style=\"background-color: rgba(255,192,203,0.3);\">";
                        }
                        else{
                            str_table = "<table style=\"background-color:pink;filter:alpha(opacity=30,Style=0)\">";
                        }

//                                                +"日期：<br>"
//                                                +"</td>"
//                                                +"</tr>"
                                                +"<tr>"
                                                +"<td style=\"position:relative;color:black;font-size:13px\">"
                                                //+"日期"

                                                +"："
                                                +aa.cell0+"<br>"
                                                +"</td>"

                                                +"<tr>"
                                                +"<td style=\"position:relative;color:black;font-size:13px\">"
