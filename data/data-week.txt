                function mouseMove()
                {
                    try
                    {
                       var ua = navigator.userAgent.toLowerCase();
                       var event = window.event || arguments.callee.caller.arguments[0];


                        if (ua.match(/Firefox/i)=="firefox")
                         {
                              if
                              (
                               event.layerX > 940 |
                               event.layerX <= 0 |
                               event.layerY <= 0
                               )
                               {
                                   return;
                               }

                         }
                         else
                         {
                              if
                              (
                               event.offsetX > 940 |
                               event.offsetX <= 0 |
                               event.offsetY <= 0
                               )
                               {
                                   return;
                               }
                           }

                        var divMoveHor = ge('divMoveHor');

                            divDetailVer = ge('divDetailVer');
                        }

                        divMoveHor.style.left = event.clientX - 53;
                        divMoveHor.style.top = 0;
//                      divMoveVer.style.left = 0;
//                      divMoveVer.style.top = event.clientY - 29;

                        if (ua.match(/Firefox/i)=="firefox")
                        {
                        currentPositionX = event.layerY;//offsetY鼠标在当前div中的坐标，clientY鼠标在整个屏幕中的坐标
                        currnetPositionY = event.layerX;
                        }
                        else

                        currnetPositionY = event.offsetX;

                        }
                        if(currnetPositionY < 150)//鼠标与左边框的距离不足div的宽度

                        {

                            divDetailVer.style.left = currnetPositionY-150;
                        }

                        var aa;

                            aa = dataObjs[event.layerX];

                        else
                        {
                            aa = dataObjs[event.offsetX];
                        }

                        }

                                                +"<td style=\"position:relative;color:black;font-size:13px\">"
                                                //+"日期"
                                                + mDateTime
                                                +"："
                                                +aa.cell0+"<br>"
                                                +"</td>"
